33  click_count_, millis(), (long)release_duration_);
34 
35  if (click_count_ == 0) {
36  // This is a new, isolated "click" that we have not yet processed.
37  click_count_++;

39  } else {

41

65 void ClickType::on_button_release() {
66  debugD(
67  "ClickType received UNPRESS for click count %d (millis: %ld, press "
68  "duration: %ld ms)",
69  click_count_, millis(), (long)press_duration_);
70 
71  if (click_count_ > 0) {
72  // This is the "release" of a click we are tracking...

74  on_ultra_long_click();

81  this->emitDelayed(ClickTypes::LongSingleClick);
82  this->on_click_completed();
83  } else if (this->click_count_ > 1) {
84  // We have just ended a double click. Sent it immediately...
85  debugD(
86  "ClickType detected DoubleClick (millis: %ld, press duration %ld ms)",
87  millis(), (long)press_duration_);
88  this->emitDelayed(ClickTypes::DoubleClick);
89  this->on_click_completed();
90  } else {
91  // This is the end of a potential single click. Queue up the send of a
92  // SingleClick report, but delay it in case another click comes in prior
93  // to the double_click_interval, which would turn this click into a
94  // DoubleClick
95  unsigned long time_of_event = millis();
96  long pd = (long)press_duration_;

99  debugD(
100  "ClickType detected SingleClick (millis: %ld, queue time: %ld, "
101  "press duration %ld ms)",
102  millis(), time_of_event, pd);
103  this->emit(ClickTypes::SingleClick);
104  this->on_click_completed();
105  });
106  }
107

110 
111  } else {

113  // an UltraLongClick has already been sent, or the producer
