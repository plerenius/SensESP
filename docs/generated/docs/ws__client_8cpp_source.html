358 void WSClient::send_access_request(const String server_address,
359  const uint16_t server_port) {
360  debugD("Preparing a new access request");
361  if (client_id == "") {
362  // generate a client ID
363  byte uuidNumber[16];
364  ESPTrueRandom.uuid(uuidNumber);
365  client_id = ESPTrueRandom.uuidToString(uuidNumber);
366  save_configuration();
367  }
368 
369  // create a new access request
370  DynamicJsonDocument doc(1024);
371  doc["clientId"] = client_id;
372  doc["description"] = String("SensESP device: ") + sensesp_app->get_hostname();
373  doc["permissions"] = this->sk_permission;
374  String json_req = "";
375  serializeJson(doc, json_req);
376 
377  HTTPClient http;
378 
379  String url = String("http://") + server_address + ":" + server_port +
380  "/signalk/v1/access/requests";
381  http.begin(wifi_client, url);
382  http.addHeader("Content-Type", "application/json");
383  int httpCode = http.POST(json_req);
384  String payload = http.getString();
385  http.end();
386 
387  // if we get a response we can't handle, try to reconnect later
388  if (httpCode != 202) {
389  debugW("Can't handle response %d to access request.", httpCode);
390  debugD("%s", payload.c_str());
391  connection_state = WSConnectionState::kWSDisconnected;
392  client_id = "";
393  return;
394  }
395 
396  // http status code 202
397 
398  deserializeJson(doc, payload.c_str());
399  String state = doc["state"];
400 
401  if (state != "PENDING") {
402  debugW("Got unknown state: %s", state.c_str());
403  connection_state = WSConnectionState::kWSDisconnected;
404  client_id = "";
405  return;
406  }
407 
408  String href = doc["href"];
409  polling_href = href;
410  save_configuration();
411 
412  debugD("Polling %s in 5 seconds", polling_href.c_str());
413  app.onDelay(5000, [this, server_address, server_port, href]() {
414  this->poll_access_request(server_address, server_port, this->polling_href);
415  });
416 }

418 void WSClient::poll_access_request(const String server_address,
419  const uint16_t server_port,
420  const String href) {
421  debugD("Polling SK Server for authentication token");
422 
423  HTTPClient http;
424 
425  String url = String("http://") + server_address + ":" + server_port + href;
426  http.begin(wifi_client, url);
427  int httpCode = http.GET();
428  if (httpCode == 200 or httpCode == 202) {
429  String payload = http.getString();
430  http.end();
431  DynamicJsonDocument doc(1024);
432  auto error = deserializeJson(doc, payload.c_str());
433  if (error) {
434  debugW("WARNING: Could not deserialize http payload.");
435  debugW("DeserializationError: %s", error.c_str());
436  return; // TODO: return at this point, or keep going?
437  }
438  String state = doc["state"];
439  debugD("%s", state.c_str());
440  if (state == "PENDING") {
441  app.onDelay(5000, [this, server_address, server_port, href]() {
442  this->poll_access_request(server_address, server_port, href);
443  });
444  return;
445  } else if (state == "COMPLETED") {
446  JsonObject access_req = doc["accessRequest"];
447  String permission =
448  access_req["permission"]; // TODO: like this in ArdJson 6? String
449  // permission =
450  // resp["accessRequest"]["permission"];
451  polling_href = "";
452  save_configuration();
453 
454  if (permission == "DENIED") {
455  debugW("Permission denied");
456  connection_state = WSConnectionState::kWSDisconnected;
457  return;
458  } else if (permission == "APPROVED") {
459  debugI("Permission granted");
460  String token = access_req["token"];
461  auth_token = token;
462  save_configuration();
463  app.onDelay(0, [this, server_address, server_port]() {
464  this->connect_ws(server_address, server_port);
465  });
466  return;
467  }
468  }
469  } else {
470  http.end();
471  if (httpCode == 500) {
472  // this is probably the server barfing due to
473  // us polling a non-existing request. Just
474  // delete the polling href.
475  debugD("Got 500, probably a non-existing request.");
476  polling_href = "";
477  save_configuration();
478  connection_state = WSConnectionState::kWSDisconnected;
479  return;
480  }
481  // any other HTTP status code
482  debugW("Can't handle response %d to pending access request.\n", httpCode);
483  connection_state = WSConnectionState::kWSDisconnected;
484  return;
485  }
